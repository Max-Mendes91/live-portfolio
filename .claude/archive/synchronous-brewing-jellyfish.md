## TECHNICAL SEO FILES

### /app/sitemap.ts
```typescript
import { MetadataRoute } from 'next';

export default function sitemap(): MetadataRoute.Sitemap {
  const baseUrl = 'https://maxmendes.dev';

  return [
    // English (default)
    { url: baseUrl, lastModified: new Date(), changeFrequency: 'weekly', priority: 1.0 },
    { url: `${baseUrl}/services`, priority: 0.9 },
    { url: `${baseUrl}/services/web-development`, priority: 0.8 },
    { url: `${baseUrl}/services/web-design`, priority: 0.8 },
    { url: `${baseUrl}/services/seo-optimization`, priority: 0.8 },
    { url: `${baseUrl}/services/ecommerce`, priority: 0.8 },
    { url: `${baseUrl}/portfolio`, priority: 0.8 },
    { url: `${baseUrl}/contact`, priority: 0.7 },

    // Polish
    { url: `${baseUrl}/pl`, priority: 1.0 },
    { url: `${baseUrl}/pl/uslugi`, priority: 0.9 },
    { url: `${baseUrl}/pl/uslugi/tworzenie-stron`, priority: 0.8 },
    { url: `${baseUrl}/pl/uslugi/projektowanie-stron`, priority: 0.8 },
    { url: `${baseUrl}/pl/uslugi/pozycjonowanie-seo`, priority: 0.8 },
    { url: `${baseUrl}/pl/uslugi/sklepy-internetowe`, priority: 0.8 },

}
```

### /app/robots.ts
```typescript
import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      { userAgent: '*', allow: '/', disallow: ['/api/', '/_next/', '/admin/'] },
    ],
    sitemap: 'https://maxmendes.dev/sitemap.xml',
    host: 'https://maxmendes.dev',
  };
}
```
